251 constexpr Matrix adjugate() const {
252 Matrix adjMat;
253 for (size i = 0; i < M; i++)
254 for (size j = 0; j < N; j++) {
255 // 构造余子式
256 Matrix<M - 1, N - 1, T> submat;
257 for (size m = 0; m < M; m++)
258 for (size n = 0; n < N; n++)
259 if (m != i && n != j)
260 submat[m > i ? m - 1 : m][n > j ? n - 1 : n] = data[m][n];
261 auto subdet = submat.det(); // 计算余子式的行列式值
