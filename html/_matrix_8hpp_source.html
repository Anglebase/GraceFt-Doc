252 constexpr Matrix adjugate() const {
253 Matrix adjMat;

255 for (size j = 0; j < N; j++) {
256 // 构造余子式
257 Matrix<M - 1, N - 1, T> submat;
258 for (size m = 0; m < M; m++)
259 for (size n = 0; n < N; n++)
260 if (m != i && n != j)
261 submat[m > i ? m - 1 : m][n > j ? n - 1 : n] = data[m][n];
262 auto subdet = submat.det(); // 计算余子式的行列式值
